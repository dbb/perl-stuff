1 #!/usr/bin/env perl 
 2 use 5.010;
 3 
 4 # https://github.com/dbbolton/
 5 #
 6 # I use this file as a reference for basic regular expressions because I have
 7 # a terrible time remembering syntax. This file is "valid" Perl in that it
 8 # can be executed without any serious errors, but the only reason for that is
 9 # proper syntax highlighting in Vim.
10 
11 
12 my $metacharacters = ' {} [] () ^ $ . | * + ? \ ' ;
13 
14 # Character classes
15 say "A Latin letter" if /[A-Za-z]/;
16 
17 # Grouping
18 say 'a is $1' if /(a)/;
19 
20 # Anchors
21 say 'Starts with "a"' if /^a/;
22 say 'Ends with "a"' if /a$/;
23 
24 # Alternation / "or"
25 say '"a" or "b"' if /a|b/;
26 
27 # Dot
28 my $anything_but_newline = '.';
29 
30 # Counts
31 say '"a" zero or more times'  if /a*/;
32 say '"a" one ore more times'  if /a+/;
33 say 'optional "a"'    if /a?/;
34 
35 say '"a" exactly 2 times' if /a{2}/;
36 say '"a" 3 to 4 times'    if /a{3,4}/;
37 say '"a" 5 or more times' if /a{5,}/;
38 
39 # abbreviations
40 say 'BOUNDARY'        if /\b/;
41 say 'not BOUNDARY'    if /\B/;
42 
43 say 'DIGIT'           if /\d/;
44 say 'not DIGIT'       if /\D/;
45 
46 say 'WHITESPACE'      if /\s/;
47 say 'not WHITESPACE'  if /\S/;
48 
49 say 'WORDCHAR'        if /\w/;
50 say 'not WORDCHAR'    if /\W/;
51 
52 # a note on '\w' and '\d': they extend beyond [A-Za-z0-9_] and [0-9]
53 # respectively (non-Roman letters and non-Arabic digits are included)
54 
55 # Backreferences
56 say '"aa"'    if /(a)\1/;
57 say '"abba"'  if /(a)(b)\2\1/;
58 say '"abba"'  if /(a)(b)\g{-1}\g{-2}/;
59 
60 my $named_backref = '(?<nom>abba)';
61 say "$+{nom}"   if /$named_backref/;
62
